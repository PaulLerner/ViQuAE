[docs]def resize(file_name, transform, output_root):
    output_path = output_root/file_name
    if output_path.exists():
        return None
    output_path.parent.mkdir(exist_ok=True, parents=True)
    image = load_image(file_name)
    if image is None:
        return None
    image = transform(image)
    save_image(image, output_path)




[docs]def batch_resize(file_names, transform, output_root):
    # hacking the way into Pool
    inputs = [(file_name, transform, output_root) for file_name in file_names]
    with Pool() as pool:
        pool.starmap(resize, inputs)
